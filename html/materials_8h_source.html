229 AWS_CRYPTOSDK_STATIC_INLINE bool aws_cryptosdk_private_refcount_down(struct aws_atomic_var *refcount) {
230  /*
231  * Memory ordering note: We must use release memory order here. Otherwise, we have the following race:
232  *
233  * Program order:
234  *
235  * Thread A:
236  * Release(obj) [if down() { free(obj) } ]
237  *
238  * Thread B:
239  * obj->foo = 1;
240  * Release(obj)
241  *
242  * Execution order:
243  *
244  * Thread B: down() -> false
245  * Thread A: down() -> true
246  * Thread A: free(obj)
247  * Thread B: obj->foo = 1
248  *
249  * To prevent this we use release order, which forbids any memory accesses coming before this point
250  * from being reordered later. This prevents thread B from reordering the obj->foo access to come after
251  * the down().
252  */
253  size_t old_count = aws_atomic_fetch_sub_explicit(refcount, 1, aws_memory_order_release);
254 
255  assert(old_count != 0);
256 
257  return old_count == 1;
258 }

263 AWS_CRYPTOSDK_STATIC_INLINE void aws_cryptosdk_private_refcount_up(struct aws_atomic_var *refcount) {
264  /*
265  * Memory ordering note: It is safe to use relaxed here. As an invariant, we assume that,
266  * on entry, the thread has some guarantee that the refcount will not reach zero until after
267  * refcount_up completes. As long as this guarantee is maintained until the next release barrier,
268  * there is no problem; if we down() on the current thread, we achieve this explicitly. Otherwise,
269  * if we communicate to some other thread that it is safe to release the reference, that communication
270  * needs to be in release order, as it's effectively a proxy down() call, for the same reasons outlined
271  * above in refcount_down().
272  *
273  * Since we've established that our initial reference won't be released until a release barrier
274  * occurs, we are happy to let the increment be deferred until that barrier occurs. We're also
275  * perfectly happy with a refcount increment happening logically earlier than expected, since this
276  * won't cause the object to be freed unexpectedly.
277  *
278  * We also note that acquire order is not required: We know that the object itself is already valid
279  * when we enter refcount_up - that is, refcount_up does not change the internal state of the object.
280  * Therefore, it's okay if some code accesses the logical state of the object "before" the refcount
281  * increment, so long as the object is not actually freed.
282  */
283  size_t old_count = aws_atomic_fetch_add_explicit(refcount, 1, aws_memory_order_relaxed);
284 
285  assert(old_count != 0 && old_count != SIZE_MAX);
286 
287  // Suppress unused variable warning when NDEBUG is set
288  (void)old_count;
289 }
